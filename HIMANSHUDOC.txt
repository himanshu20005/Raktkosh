CREATE TABLE DONOR
(
	NAME VARCHAR(30),
	SEX VARCHAR(1),
	ADDRESS VARCHAR(30),
	AGE INT,
	PHONE_NO BIGINT,
	D_ID INT PRIMARY KEY
);

CREATE TABLE EMPLOYEE
(
	NAME VARCHAR(30),
	ADDRESS VARCHAR(30),
	PHONE_NO BIGINT,
	E_ID INT PRIMARY KEY	
);

CREATE TABLE BLOOD
(
	TYPE VARCHAR(10),
	I_NO INT,
	B_ID INT PRIMARY KEY
);

CREATE TABLE INVENTORY
(
	D_ID INT REFERENCES DONOR(D_ID) ON DELETE CASCADE,
	I_NO INT PRIMARY KEY,
	B_ID INT REFERENCES BLOOD(B_ID) ON DELETE CASCADE,
	TYPE VARCHAR(10)
);

CREATE TABLE MANAGER
(
	NAME VARCHAR(30),
	ADDRESS VARCHAR(30),
	CONTACT_NO BIGINT,
	I_NO INT REFERENCES INVENTORY(I_NO) ON DELETE CASCADE,
	E_ID INT REFERENCES EMPLOYEE(E_ID) ON DELETE CASCADE
);

CREATE TABLE REQUESTER
(
	NAME VARCHAR(30),
	ADDRESS VARCHAR(30),
	PHONE_NO BIGINT,
	I_NO INT REFERENCES INVENTORY(I_NO) ON DELETE CASCADE,
	TYPE VARCHAR(10)
);
